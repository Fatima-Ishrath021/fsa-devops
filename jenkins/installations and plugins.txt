#TOMCAT ISNTALLATIONS

Update repo
>> sudo apt update -y

install tomcat10
>> sudo apt install tomcat10 -y

install tomcat10-admin
>> sudo apt install tomcat10-admin  -y
>> sudo apt update -y
>> sudo systemctl restart tomcat10

EDIT Users-xml file
>> sudo nano /etc/tomcat10/tomcat-users.xml
	addd==> <user username="admin" password="admin" roles="manager-gui, 	manager-script,manager-status"/>
>> sudo systemctl restart tomcat10
